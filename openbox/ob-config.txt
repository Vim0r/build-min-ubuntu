====How to reboot/poweroff in openbox without passwd=======
  sudo visudo
## add:
ALL   ALL=NOPASSWD:/sbin/reboot
ALL   ALL=NOPASSWD:/sbin/poweroff
  nano ~/.config/openbox/menu.xml
## add:
		<separator label="Logout"/>
		<item label="ReBoot">
			<action name="Execute">
				<execute>sudo /sbin/reboot</execute>
			</action>
		</item>
		<item label="PowerOff">
			<action name="Execute">
				<execute>sudo /sbin/poweroff</execute>
			</action>
		</item>

# set keybind
# =======add in /home/vm/.config/openbox/rc.xml==========
    <keybind key="F12">
      <action name="Execute">

	<keybind key="C-A-space">
